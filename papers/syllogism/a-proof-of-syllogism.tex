\maketitle

\infer
{
  (\beta \rightarrow \gamma)
  \rightarrow (\alpha \rightarrow \beta)
  \rightarrow \alpha
  \rightarrow \gamma
}
{
  \infer
  {
    (\alpha \rightarrow \beta)
    \rightarrow \alpha
    \rightarrow \gamma
  }
  {
    \infer
    {
      \alpha \rightarrow \gamma
    }
    {
      \infer
      {
        \gamma
      }
      {
        \infer
        {
          \beta
        }
        {
          \alpha \rightarrow \beta
          &
          \alpha
        }
        &
        \beta \rightarrow \gamma
      }
    }
  }
}
